      React hook for using Strata's persistent storage API. Provides a convenient interface for saving and loading game data.

        Optional namespace to isolate game data (default: 'strata')

      Storage utilities for get, set, remove, and clear operations

      • clearAllSaves: () => Promise<void>

        Clear all game data in the namespace.

      Example

      function GameComponent() {
        const { saveGame, loadGame } = useStorage('mygame');

        const handleSave = async () => {
          await saveGame('progress', { level: 5, score: 1000 });
        };

        const handleLoad = async () => {
          const { value } = await loadGame<{ level: number; score: number }>('progress');
          if (value) {
            console.log(`Resuming at level ${value.level}`);
          }
        };
      }
